## 一、引擎启动流程

```
main.rs
│
├─ 1. 加载 .env 配置
│
├─ 2. 初始化存储层 (redb)
│
├─ 3. 加载数据文件 (data/characters/*.json, data/world/*.json, data/plot/*.json)
│   └─ → 构建 WorldState
│
├─ 4. 启动 Axum HTTP 服务器
│   ├─ 注册 REST API 路由
│   └─ 注册 WebSocket 路由
│
├─ 5. 初始化模拟引擎
│   ├─ 创建 LLM Gateway
│   ├─ 初始化 Simulation Loop
│   └─ 设置初始速度模式 (Paused)
│
└─ 6. 等待用户通过 Dashboard 发送控制指令
    ├─ POST /api/v1/simulation/start
    ├─ POST /api/v1/simulation/pause
    ├─ POST /api/v1/simulation/speed  (fast/detailed)
    └─ POST /api/v1/simulation/step   (单步推进)
```

**启动不自动跑模拟**。默认状态是 `Paused`，等待用户从 Dashboard 发出指令。

```rust
pub async fn run_simulation(engine: Arc<Mutex<SimulationEngine>>) {
    loop {
        let state = { engine.lock().state.clone() };
        
        match state {
            EngineState::Stopped => break,
            EngineState::Paused => {
                tokio::time::sleep(Duration::from_millis(100)).await;
                continue;
            }
            EngineState::Running { speed, .. } => {
                match speed {
                    SimSpeed::FastForward => {
                        fastforward_tick(&mut engine).await;
                    }
                    SimSpeed::Detailed => {
                        detailed_tick(&mut engine).await;
                    }
                }
                // tick 间暂停（防止 100% CPU）
                tokio::time::sleep(Duration::from_millis(50)).await;
            }
        }
    }
}
```

### 2.3 FastForward Tick（零 LLM 调用）

```rust
async fn fastforward_tick(engine: &mut SimulationEngine) {
    // 1. 更新时间
    engine.world.timeline.advance_hours(1);
    
    // 2. 更新环境（温度/天气/季节）
    engine.world.environment.update(&engine.world.timeline);
    
    // 3. 资源消耗（每地点按人口扣减）
    for location in &mut engine.world.locations {
        location.tick_resource_consumption();
    }
    
    // 4. 角色生理衰减（hecs 批量遍历）
    let mut query = engine.world.characters.query::<&mut CharacterState>();
    for char_state in query.iter() {
        char_state.tick_physiology(&engine.world.environment);
    }
    
    // 5. bhava 情绪自然衰减
    engine.world.tick_all_affects();
    
    // 6. 检查条件事件触发器
    let triggered = engine.world.event_system.check_conditional();
    if !triggered.is_empty() {
        // 有事件触发 → 自动切换到 Detailed 模式
        engine.switch_to_detailed();
        engine.ws_broadcaster.send(EngineEvent::SpeedChanged {
            from: SimSpeed::FastForward,
            to: SimSpeed::Detailed,
            reason: format!("条件事件触发: {:?}", triggered),
        });
    }
    
    // 7. 统计
    engine.stats.fastforward_ticks += 1;
    engine.stats.total_ticks += 1;
    
    // 8. 检查是否需要保存检查点
    if engine.stats.total_ticks % engine.config.checkpoint_interval == 0 {
        save_checkpoint(&engine).await;
    }
    
    // 9. 广播状态给前端
    engine.ws_broadcaster.send(EngineEvent::Tick {
        tick: engine.stats.total_ticks,
        time: engine.world.timeline.now.clone(),
        speed: SimSpeed::FastForward,
    });
}
```

### 2.4 Detailed Tick（有 LLM 调用）

```rust
async fn detailed_tick(engine: &mut SimulationEngine) {
    // 1. 更新环境（同 FF，但粒度细）
    engine.world.timeline.advance_minutes(5);
    engine.world.environment.update(&engine.world.timeline);
    
    // 2. 感知阶段：每个角色感知环境，确定是否需要 LLM 决策
    let mut need_llm = Vec::new();
    let mut idle_count = 0;
    
    let mut query = engine.world.characters.query::<(&CharacterState, &mut CharacterActivity)>();
    for (state, activity) in query.iter() {
        activity.perceive(&engine.world);  // 感知更新
        if activity.needs_llm_decision() {
            need_llm.push(activity.entity_id());
        } else {
            idle_count += 1;
        }
    }
    
    // 3. 检查空闲阈值 → 自动切 FastForward
    let total_chars = engine.world.characters.query::<&CharacterState>().iter().count() as f64;
    if idle_count as f64 / total_chars > 0.8
        && engine.consecutive_idle >= engine.config.idle_threshold
    {
        engine.switch_to_fastforward();
        engine.ws_broadcaster.send(EngineEvent::SpeedChanged {
            from: SimSpeed::Detailed,
            to: SimSpeed::FastForward,
            reason: "大部分角色无行动".into(),
        });
        return;  // 本次 tick 结束，不调 LLM
    }
    
    // 4. 并行 LLM 决策
    if !need_llm.is_empty() {
        let decisions = engine.llm_gateway
            .batch_decide(&engine.world, &need_llm)
            .await;
        
        // 5. 应用决策
        for decision in decisions {
            apply_decision(&mut engine.world, decision);
        }
        
        // 6. 情绪引擎更新
        engine.world.tick_all_affects();
        
        // 7. 关系图自动更新（根据交互事件）
        engine.world.relationship_graph.update_from_events();
        
        engine.stats.llm_calls_this_step = need_llm.len();
    }
    
    // 8. 资源消耗（同 FF）
    for location in &mut engine.world.locations {
        location.tick_resource_consumption();
    }
    
    // 9. 统计
    engine.stats.detailed_ticks += 1;
    engine.stats.total_ticks += 1;
    
    // 10. 检查点
    if engine.stats.total_ticks % engine.config.checkpoint_interval == 0 {
        save_checkpoint(&engine).await;
    }
    
    // 11. 广播
    engine.ws_broadcaster.send(EngineEvent::DetailedTick {
        tick: engine.stats.total_ticks,
        time: engine.world.timeline.now.clone(),
        decisions: decisions_summary,
        events: triggered_events,
    });
}
```

### 切换逻辑实现

```rust
impl SimulationEngine {
    fn check_speed_transition(&mut self) -> Option<SpeedTransition> {
        match self.state {
            EngineState::Running { speed: SimSpeed::Detailed, .. } => {
                // 降速条件：连续 idle >= threshold
                if self.consecutive_idle >= self.config.idle_threshold {
                    self.consecutive_idle = 0;
                    return Some(SpeedTransition {
                        from: SimSpeed::Detailed,
                        to: SimSpeed::FastForward,
                        reason: format!("连续 {} tick 无活跃事件", self.consecutive_idle),
                    });
                }
            }
            EngineState::Running { speed: SimSpeed::FastForward, .. } => {
                // 升速条件：事件队列有到期事件 / 条件触发
                if let Some(events) = self.world.event_system.peek_due() {
                    self.consecutive_idle = 0;
                    return Some(SpeedTransition {
                        from: SimSpeed::FastForward,
                        to: SimSpeed::Detailed,
                        reason: format!("{} 个事件待处理", events.len()),
                    });
                }
            }
            _ => {}
        }
        None
    }
}
```

### 4.3 三级 Prompt 链

每个角色的决策分三步，每步都是独立的 LLM 调用（结构化 JSON 输出）：

```
Step 1: 感知 (Perceive)
  system: 角色卡（性格/记忆/当前状态）
  user:   场景描述 + 角色感知范围内的信息
  └→ output: { noticed: [...], priority_level: high/normal/low }

Step 2: 思考 (Think)
  system: 角色卡
  user:   感知结果 + 当前计划 + 可选行动列表
  └→ output: { thought: "...", options: [{ action: "...", reason: "..." }], chosen: 0 }

Step 3: 行动 (Act)
  system: 角色卡
  user:   选择的行动
  └→ output: { dialogue: "...", action: "...", expression: "...", inner_thought: "..." }
```

**为什么不一次调用搞定？**
1. 每步输出结构化 JSON，可校验
2. 可以在 Step 2 停住（角色在思考但未行动），降低 30% API 成本
3. 行动层的 prompt 更短（不需要感知上下文），减少 token 消耗
4. 便于调试：可以查看角色"想了什么"和"做了什么"的差异

```rust
pub struct ScheduledEvent {
    pub id: Uuid,
    pub trigger_time: SimTime,       // 触发时间
    pub event_type: EventType,
    pub content: String,              // 事件描述
    pub participants: Vec<Uuid>,      // 参与者
    pub priority: EventPriority,      // 优先级
    pub source: EventSource,          // 来源（预定/条件/连锁）
    pub cascade: Option<CascadeRule>, // 连锁规则
}

// BinaryHeap 需要按时间排序（最早的优先）
impl Ord for ScheduledEvent {
    fn cmp(&self, other: &Self) -> Ordering {
        // 注意：BinaryHeap 是 "最大堆"，所以用 reverse
        other.trigger_time.cmp(&self.trigger_time)
            .then_with(|| other.priority.cmp(&self.priority))
    }
}
```

```rust
impl CheckpointManager {
    /// 保存全量快照
    pub fn save_snapshot(&self, world: &WorldState, tag: &str) -> Result<()> {
        let data = match self.save_format {
            SaveFormat::Bincode => bincode::serialize(world)?,
            SaveFormat::Json => serde_json::to_vec(world)?,
        };
        let path = self.base_dir.join("auto")
            .join(format!("tick_{:06}_{}.bin", world.tick_count, tag));
        std::fs::write(&path, &data)?;
        
        // redb 记录索引
        let tx = self.db.begin_write()?;
        tx.insert("checkpoints", tag, CheckpointMeta {
            path, tick: world.tick_count, time: world.timeline.now.clone(),
        })?;
        tx.commit()?;
        
        Ok(())
    }
    
    /// 加载指定检查点
    pub fn load_snapshot(&self, tag: &str) -> Result<WorldState> {
        let tx = self.db.begin_read()?;
        let meta = tx.get("checkpoints", tag)?
            .ok_or(Error::CheckpointNotFound(tag.to_string()))?;
        let data = std::fs::read(&meta.path)?;
        Ok(bincode::deserialize(&data)?)
    }
}
```

## 附一、叙事议会集成

> 实现详见 `13-叙事议会与标签系统.md`

模拟主循环每 50 tick 异步触发一轮叙事议会。议会不阻塞主循环——Rust 继续跑后续 tick，议会结果到了再写回参数。

| 集成点 | 位置 | 影响 |
|-------|------|------|
| 议会触发 | 每个 Detailed Tick 末尾 | 异步，不阻塞 |
| 指令写回 | tokio::spawn 回调中 | 更新 WorldState.narrative_directive |
| Storyteller 读取 | 每 tick 开始 | 调整 event_type_weights、target_tension |
| EmergentDetector 读取 | 碰撞检测前 | 调整 collision_sensitivity |
| ProbabilityTree 读取 | 分支展开前 | 调整分支概率偏移 |
| TagIndex 读取 | 信息维护时 | 复活/归档线索 |

```rust
/// 主循环末尾的议会触发
pub async fn simulation_loop(engine: &mut SimulationEngine) {
    loop {
        // ... 现有循环逻辑 ...
        
        // 议会触发（每 50 tick）
        if engine.stats.total_ticks % 50 == 0 
            && engine.stats.last_council_tick != engine.stats.total_ticks 
        {
            engine.stats.last_council_tick = engine.stats.total_ticks;
            let input = CouncilInputBuilder::build(&engine.world);
            let engine_clone = engine.clone();
            tokio::spawn(async move {
                let directive = NarrativeCouncil::convene(input).await;
                engine_clone.apply_narrative_directive(directive).await;
            });
        }
    }
}
```
